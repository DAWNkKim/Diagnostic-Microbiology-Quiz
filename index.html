let currentData = null; // 현재 선택된 과목의 데이터 let currentSubjectKey = 'micro'; // 현재 과목 키 (파일명과 일치해야 함) let wrongAnswers = JSON.parse(localStorage.getItem('wrongAnswers_v4')) || {}; // 1. 탭을 누르면 해당 JSON 파일을 서버(깃허브)에서 가져옵니다. async function loadSubject(subjectKey) { currentSubjectKey = subjectKey; try { // 파일명: micro.json 또는 chem.json 등을 호출 const response = await fetch(`${subjectKey}.json`); currentData = await response.json(); // 화면 업데이트 updateTabUI(subjectKey); renderChapters(); updateInfo(); } catch (error) { alert(`${subjectKey}.json 파일을 찾을 수 없습니다.`); } } function updateTabUI(key) { document.querySelectorAll('.tab').forEach(t => { // 탭의 onclick 속성에 적힌 키값과 비교 t.classList.toggle('active', t.getAttribute('onclick').includes(key)); }); document.getElementById('subject-title').innerText = `${currentData.name} 챕터 선택`; } function renderChapters() { const list = document.getElementById('chapter-list'); list.innerHTML = ''; Object.keys(currentData.chapters).forEach(ch => { const div = document.createElement('div'); div.className = 'chapter-item'; div.innerText = ch; div.onclick = () => { div.classList.toggle('selected'); if(selectedChapters.has(ch)) selectedChapters.delete(ch); else selectedChapters.add(ch); }; list.appendChild(div); }); } // 시험 생성 시 현재 불러온 currentData를 사용 function generateExam(mode) { examPaper = []; if (mode === 'wrong') { const subWrongs = wrongAnswers[currentSubjectKey] || []; if (subWrongs.length === 0) { alert("저장된 오답이 없습니다!"); return; } examPaper = [...subWrongs].sort(() => 0.5 - Math.random()); } else { if (selectedChapters.size === 0) { alert("챕터를 선택해주세요."); return; } selectedChapters.forEach(ch => { const data = currentData.chapters[ch]; if (mode === 'ratio') { examPaper = examPaper.concat(getRandom(data.basic, 18), getRandom(data.scenario, 6)); } else { examPaper = examPaper.concat(data.basic, data.scenario); } }); examPaper.sort(() => 0.5 - Math.random()); } // ... 이하 출력 로직 동일 }
